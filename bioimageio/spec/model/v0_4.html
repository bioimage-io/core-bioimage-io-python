 495        if key == "keras_hdf5":
 496            ret = self.keras_hdf5

 498            ret = self.onnx
 499        elif key == "pytorch_state_dict":
 500            ret = self.pytorch_state_dict
 501        elif key == "tensorflow_js":
 502            ret = self.tensorflow_js
