172def _get_stat(
173    model_descr: AnyModelDescr,
174    dataset: Iterable[Sample],
175    dataset_length: int,
176    stats_path: Path,
177) -> Mapping[DatasetMeasure, MeasureValue]:
178    req_dataset_meas, _ = get_required_dataset_measures(model_descr)
179    if not req_dataset_meas:
180        return {}
181
182    req_dataset_meas, _ = get_required_dataset_measures(model_descr)
183
184    if stats_path.exists():
185        logger.info(f"loading precomputed dataset measures from {stats_path}")
186        stat = load_dataset_stat(stats_path)
187        for m in req_dataset_meas:
188            if m not in stat:
189                raise ValueError(f"Missing {m} in {stats_path}")
190
191        return stat
192
193    stats_calc = StatsCalculator(req_dataset_meas)
194
195    for sample in tqdm(
196        dataset, total=dataset_length, desc="precomputing dataset stats", unit="sample"
197    ):
198        stats_calc.update(sample)
199
200    stat = stats_calc.finalize()
201    save_dataset_stat(stat, stats_path)
202
203    return stat

375    def run(self):
376        if self.example:
377            return self._example()
378
379        model_descr = ensure_description_is_model(self.descr)
380
381        input_ids = get_member_ids(model_descr.inputs)
382        output_ids = get_member_ids(model_descr.outputs)
383
384        minimum_input_ids = tuple(
385            str(ipt.id) if isinstance(ipt, v0_5.InputTensorDescr) else str(ipt.name)
386            for ipt in model_descr.inputs
387            if not isinstance(ipt, v0_5.InputTensorDescr) or not ipt.optional
388        )
389        maximum_input_ids = tuple(
390            str(ipt.id) if isinstance(ipt, v0_5.InputTensorDescr) else str(ipt.name)
391            for ipt in model_descr.inputs
392        )
393
394        def expand_inputs(i: int, ipt: Union[str, Tuple[str, ...]]) -> Tuple[str, ...]:
395            if isinstance(ipt, str):
396                ipts = tuple(
397                    ipt.format(model_id=self.descr_id, input_id=t) for t in input_ids
398                )
399            else:
400                ipts = tuple(
401                    p.format(model_id=self.descr_id, input_id=t)
402                    for t, p in zip(input_ids, ipt)
403                )
404
405            if len(set(ipts)) < len(ipts):
406                if len(minimum_input_ids) == len(maximum_input_ids):
407                    n = len(minimum_input_ids)
408                else:
409                    n = f"{len(minimum_input_ids)}-{len(maximum_input_ids)}"
410
411                raise ValueError(
412                    f"[input sample #{i}] Include '{{input_id}}' in path pattern or explicitly specify {n} distinct input paths (got {ipt})"
413                )
414
415            if len(ipts) < len(minimum_input_ids):
416                raise ValueError(
417                    f"[input sample #{i}] Expected at least {len(minimum_input_ids)} inputs {minimum_input_ids}, got {ipts}"
418                )
419
420            if len(ipts) > len(maximum_input_ids):
421                raise ValueError(
422                    f"Expected at most {len(maximum_input_ids)} inputs {maximum_input_ids}, got {ipts}"
423                )
424
425            return ipts
426
427        inputs = [expand_inputs(i, ipt) for i, ipt in enumerate(self.inputs, start=1)]
428
429        sample_paths_in = [
430            {t: Path(p) for t, p in zip(input_ids, ipts)} for ipts in inputs
431        ]
432
433        sample_ids = _get_sample_ids(sample_paths_in)
434
435        def expand_outputs():
436            if isinstance(self.outputs, str):
437                outputs = [
438                    tuple(
439                        Path(
440                            self.outputs.format(
441                                model_id=self.descr_id, output_id=t, sample_id=s
442                            )
443                        )
444                        for t in output_ids
445                    )
446                    for s in sample_ids
447                ]
448            else:
449                outputs = [
450                    tuple(
451                        Path(p.format(model_id=self.descr_id, output_id=t, sample_id=s))
452                        for t, p in zip(output_ids, self.outputs)
453                    )
454                    for s in sample_ids
455                ]
456
457            for i, out in enumerate(outputs, start=1):
458                if len(set(out)) < len(out):
459                    raise ValueError(
460                        f"[output sample #{i}] Include '{{output_id}}' in path pattern or explicitly specify {len(output_ids)} distinct output paths (got {out})"
461                    )
462
463                if len(out) != len(output_ids):
464                    raise ValueError(
465                        f"[output sample #{i}] Expected {len(output_ids)} outputs {output_ids}, got {out}"
466                    )
467
468            return outputs
469
470        outputs = expand_outputs()
471
472        sample_paths_out = [
473            {MemberId(t): Path(p) for t, p in zip(output_ids, out)} for out in outputs
474        ]
475
476        if not self.overwrite:
477            for sample_paths in sample_paths_out:
478                for p in sample_paths.values():
479                    if p.exists():
480                        raise FileExistsError(
481                            f"{p} already exists. use --overwrite to (re-)write outputs anyway."
482                        )
483        if self.preview:
484            print("🛈 bioimageio prediction preview structure:")
485            pprint(
486                {
487                    "{sample_id}": dict(
488                        inputs={"{input_id}": "<input path>"},
489                        outputs={"{output_id}": "<output path>"},
490                    )
491                }
492            )
493            print("🔎 bioimageio prediction preview output:")
494            pprint(
495                {
496                    s: dict(
497                        inputs={t: p.as_posix() for t, p in sp_in.items()},
498                        outputs={t: p.as_posix() for t, p in sp_out.items()},
499                    )
500                    for s, sp_in, sp_out in zip(
501                        sample_ids, sample_paths_in, sample_paths_out
502                    )
503                }
504            )
505            return
506
507        def input_dataset(stat: Stat):
508            for s, sp_in in zip(sample_ids, sample_paths_in):
509                yield load_sample_for_model(
510                    model=model_descr,
511                    paths=sp_in,
512                    stat=stat,
513                    sample_id=s,
514                )
515
516        stat: Dict[Measure, MeasureValue] = dict(
517            _get_stat(
518                model_descr, input_dataset({}), len(sample_ids), self.stats
519            ).items()
520        )
521
522        pp = create_prediction_pipeline(
523            model_descr,
524            weight_format=None if self.weight_format == "any" else self.weight_format,
525        )
526        predict_method = (
527            pp.predict_sample_with_blocking
528            if self.blockwise
529            else pp.predict_sample_without_blocking
530        )
531
532        for sample_in, sp_out in tqdm(
533            zip(input_dataset(dict(stat)), sample_paths_out),
534            total=len(inputs),
535            desc=f"predict with {self.descr_id}",
536            unit="sample",
537        ):
538            sample_out = predict_method(sample_in)
539            save_sample(sp_out, sample_out)

632def _get_sample_ids(
633    input_paths: Sequence[Mapping[MemberId, Path]]
634) -> Sequence[SampleId]:
635    """Get sample ids for given input paths, based on the common path per sample.
636
637    Falls back to sample01, samle02, etc..."""
638
639    matcher = SequenceMatcher()
640
641    def get_common_seq(seqs: Sequence[Sequence[str]]) -> Sequence[str]:
642        """extract a common sequence from multiple sequences
643        (order sensitive; strips whitespace and slashes)
644        """
645        common = seqs[0]
646
647        for seq in seqs[1:]:
648            if not seq:
649                continue
650            matcher.set_seqs(common, seq)
651            i, _, size = matcher.find_longest_match()
652            common = common[i : i + size]
653
654        if isinstance(common, str):
655            common = common.strip().strip("/")
656        else:
657            common = [cs for c in common if (cs := c.strip().strip("/"))]
658
659        if not common:
660            raise ValueError(f"failed to find common sequence for {seqs}")
661
662        return common
663
664    def get_shorter_diff(seqs: Sequence[Sequence[str]]) -> List[Sequence[str]]:
665        """get a shorter sequence whose entries are still unique
666        (order sensitive, not minimal sequence)
667        """
668        min_seq_len = min(len(s) for s in seqs)
669        # cut from the start
670        for start in range(min_seq_len - 1, -1, -1):
671            shortened = [s[start:] for s in seqs]
672            if len(set(shortened)) == len(seqs):
673                min_seq_len -= start
674                break
675        else:
